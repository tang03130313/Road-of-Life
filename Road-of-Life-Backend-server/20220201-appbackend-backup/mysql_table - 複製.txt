create table patient(id int(11) unsigned not null auto_increment, name varchar(10) not null, phone varchar(10) not null,email varchar(40) not null,password varchar(15) not null,user_id varchar(11) default null,PRIMARY KEY(id),index index_name(name),index index_email(email),index index_userid(user_id))
character set = utf8
PARTITION BY HASH(id)
PARTITIONS 10;
